007import java.util.HashMap;
008import java.util.HashSet;
009import java.util.Map;
010import java.util.Set;
011import java.util.SortedMap;
012import java.util.SortedSet;
013import java.util.TreeMap;
014import java.util.TreeSet;
015
016import org.codehaus.groovy.grails.commons.DomainClassArtefactHandler;
017import org.codehaus.groovy.grails.commons.GrailsApplication;
018import org.codehaus.groovy.grails.commons.GrailsClassUtils;
019import org.codehaus.groovy.grails.commons.GrailsDomainClass;
020import org.codehaus.groovy.grails.commons.GrailsDomainClassProperty;
021import org.codehaus.groovy.grails.support.proxy.DefaultProxyHandler;
022import org.codehaus.groovy.grails.support.proxy.EntityProxyHandler;
023import org.codehaus.groovy.grails.support.proxy.ProxyHandler;
024import org.codehaus.groovy.grails.web.converters.ConverterUtil;
025import org.codehaus.groovy.grails.web.converters.exceptions.ConverterException;
026import org.codehaus.groovy.grails.web.converters.marshaller.ObjectMarshaller;
027import org.codehaus.groovy.grails.web.json.JSONWriter;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030import org.springframework.beans.BeanWrapper;
031import org.springframework.beans.BeanWrapperImpl;

033/**
034 * @author Siegfried Puchbauer, Eduardo Goncalves
035 * @since 1.1
036 */
037public class CustomDomainMarshaller implements ObjectMarshaller<JSON> {
038
039    private boolean includeVersion = false;
040    private ProxyHandler proxyHandler;
041    private GrailsApplication application;
042
043    protected final Logger log = LoggerFactory.getLogger(getClass());
044    
045    public CustomDomainMarshaller(boolean includeVersion, GrailsApplication application) {
046        this(includeVersion, new DefaultProxyHandler(), application);
047    }
048
049    public CustomDomainMarshaller(boolean includeVersion, ProxyHandler proxyHandler, GrailsApplication application) {
050        this.includeVersion = includeVersion;
051        this.proxyHandler = proxyHandler;
052        this.application = application;
053    }
054
055    public boolean isIncludeVersion() {
056        return includeVersion;
057    }
058
059    public void setIncludeVersion(boolean includeVersion) {
060        this.includeVersion = includeVersion;
061    }
062
063    public boolean supports(Object object) {
064        String name = ConverterUtil.trimProxySuffix(object.getClass().getName());
065        log.info("Marshalling object from json ");
066        return application.isArtefactOfType(DomainClassArtefactHandler.TYPE, name);
067
068    }
069
070    @SuppressWarnings({ "unchecked", "rawtypes" })
071    public void marshalObject(Object value, JSON json) throws ConverterException {
072        log.info("Marshalling json from object " + value);
073        JSONWriter writer = json.getWriter();
074        value = proxyHandler.unwrapIfProxy(value);
075        Class<?> clazz = value.getClass();
076
077        GrailsDomainClass domainClass = (GrailsDomainClass)application.getArtefact(
078              DomainClassArtefactHandler.TYPE, ConverterUtil.trimProxySuffix(clazz.getName()));
079        BeanWrapper beanWrapper = new BeanWrapperImpl(value);
080
081        writer.object();
082        writer.key("class").value(domainClass.getClazz().getName());
083
084        GrailsDomainClassProperty id = domainClass.getIdentifier();
085        Object idValue = extractValue(value, id);
086
087        json.property("id", idValue);
088
089        if (isIncludeVersion()) {
090            GrailsDomainClassProperty versionProperty = domainClass.getVersion();
091            Object version = extractValue(value, versionProperty);
092            json.property("version", version);
093        }
094
095        GrailsDomainClassProperty[] properties = domainClass.getPersistentProperties();
096
097        for (GrailsDomainClassProperty property : properties) {
098            writer.key(property.getName());
099            if (!property.isAssociation()) {
100                // Write non-relation property
101                Object val = beanWrapper.getPropertyValue(property.getName());
102                json.convertAnother(val);
103            }
104            else {
105                Object referenceObject = beanWrapper.getPropertyValue(property.getName());
106                if (isRenderDomainClassRelations()) {
107                    if (referenceObject == null) {
108                        writer.value(null);
109                    }
110                    else {
111                        referenceObject = proxyHandler.unwrapIfProxy(referenceObject);
112                        if (referenceObject instanceof SortedMap) {
113                            referenceObject = new TreeMap((SortedMap) referenceObject);
114                        }
115                        else if (referenceObject instanceof SortedSet) {
116                            referenceObject = new TreeSet((SortedSet) referenceObject);
117                        }
118                        else if (referenceObject instanceof Set) {
119                            referenceObject = new HashSet((Set) referenceObject);
120                        }
121                        else if (referenceObject instanceof Map) {
122                            referenceObject = new HashMap((Map) referenceObject);
123                        }
124                        else if (referenceObject instanceof Collection) {
125                            referenceObject = new ArrayList((Collection) referenceObject);
126                        }
127                        json.convertAnother(referenceObject);
128                    }
129                }
130                else {
131                    if (referenceObject == null) {
132                        json.value(null);
133                    }
134                    else {
135                        GrailsDomainClass referencedDomainClass = property.getReferencedDomainClass();
136
137                        // Embedded are now always fully rendered
138                        if (referencedDomainClass == null || property.isEmbedded() || GrailsClassUtils.isJdk5Enum(property.getType())) {
139                            json.convertAnother(referenceObject);
140                        }
141                        else if (property.isOneToOne() || property.isManyToOne() || property.isEmbedded()) {
142                            asShortObject(referenceObject, json, referencedDomainClass.getIdentifier(), referencedDomainClass);
143                        }
144                        else {
145                            GrailsDomainClassProperty referencedIdProperty = referencedDomainClass.getIdentifier();
146                            @SuppressWarnings("unused")
147                            String refPropertyName = referencedDomainClass.getPropertyName();
148                            if (referenceObject instanceof Collection) {
149                                Collection o = (Collection) referenceObject;
150                                writer.array();
151                                for (Object el : o) {
152                                    asShortObject(el, json, referencedIdProperty, referencedDomainClass);
153                                }
154                                writer.endArray();
155                            }
156                            else if (referenceObject instanceof Map) {
157                                Map<Object, Object> map = (Map<Object, Object>) referenceObject;
158                                for (Map.Entry<Object, Object> entry : map.entrySet()) {
159                                    String key = String.valueOf(entry.getKey());
160                                    Object o = entry.getValue();
161                                    writer.object();
162                                    writer.key(key);
163                                    asShortObject(o, json, referencedIdProperty, referencedDomainClass);
164                                    writer.endObject();
165                                }
166                            }
167                        }
168                    }
169                }
170            }
171        }
172        writer.endObject();
173    }
174
175    protected void asShortObject(Object refObj, JSON json, GrailsDomainClassProperty idProperty, GrailsDomainClass referencedDomainClass) throws ConverterException {
176
177        Object idValue;
178        log.info("Marshalling object from json ");
179        
180        if (proxyHandler instanceof EntityProxyHandler) {
181            idValue = ((EntityProxyHandler) proxyHandler).getProxyIdentifier(refObj);
182            if (idValue == null) {
183                idValue = extractValue(refObj, idProperty);
184            }
185        }
186        else {
187            idValue = extractValue(refObj, idProperty);
188        }
189        JSONWriter writer = json.getWriter();
190        writer.object();
191        writer.key("class").value(referencedDomainClass.getName());
192        writer.key("id").value(idValue);
193        writer.endObject();
194    }
195
196    protected Object extractValue(Object domainObject, GrailsDomainClassProperty property) {
197        BeanWrapper beanWrapper = new BeanWrapperImpl(domainObject);
198        log.info("Marshalling object from json ");
199        return beanWrapper.getPropertyValue(property.getName());
200    }
201
202    protected boolean isRenderDomainClassRelations() {
203        log.info("Marshalling object from json ");
204        return false;
205    }
206}
